public class ResponseResult implements Serializable{

    private int code; //
    private String message; //
    private CourseWorkCountVO[] data; //

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public CourseWorkCountVO[] getData(){
        return data;
    }

    public void setData(CourseWorkCountVO[] data){
        this.data = data;
    }

}

public class CourseWorkCountVO implements Serializable{

    private long id; //课程类目id
    private String courseCateName; //课程类目名称
    private long waitCorrectWorkCount; //待批改的作业数量

    public long getId(){
        return id;
    }

    public void setId(long id){
        this.id = id;
    }

    public String getCourseCateName(){
        return courseCateName;
    }

    public void setCourseCateName(String courseCateName){
        this.courseCateName = courseCateName;
    }

    public long getWaitCorrectWorkCount(){
        return waitCorrectWorkCount;
    }

    public void setWaitCorrectWorkCount(long waitCorrectWorkCount){
        this.waitCorrectWorkCount = waitCorrectWorkCount;
    }

}
