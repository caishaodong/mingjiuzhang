public class ResponseResult implements Serializable{

    private int code; //
    private String message; //
    private ActivityRequest[] data; //

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public ActivityRequest[] getData(){
        return data;
    }

    public void setData(ActivityRequest[] data){
        this.data = data;
    }

}
