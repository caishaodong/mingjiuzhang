获取用户信息

作者: caishaodong

请求URL

api/user/getInfo GET

返回结果

{
	"code":"int",
	"message":"string",
	"data":{
		"log":{},
		"id":"long //用户ID",
		"identity":"string //用户身份：teacher老师，student学生",
		"username":"string //昵称",
		"realName":"string //真实姓名",
		"email":"string //电子邮件",
		"mobile":"string //手机号码",
		"cardId":"string //证件ID",
		"avatar":"string //图像url",
		"studyNumber":"string //学号",
		"grade":"string //年级（0其他，1一年级，2二年级，3三年级，4四年级，5五年级，6六年级，7七年级，8八年级）",
		"sex":"int //性别 1:男,2:女",
		"provinceCode":"int //省编码",
		"cityCode":"int //市编码",
		"areaCode":"int //区编码",
		"province":"string //省",
		"city":"string //市",
		"area":"string //区",
		"totalIntegral":"long //用户历史总积分",
		"integral":"long //用户当前积分",
		"isMember":"int //是否是会员：0不是  1是",
		"userImage":"string //商户地址",
		"isDeleted":"int //是否删除：0否 1是",
		"gmtCreate":"date //创建时间",
		"gmtModified":"date //更新时间"
	}
}

修改用户信息

作者: caishaodong

请求URL

api/user/update PUT

请求体 application/json

{
	"type":"string //修改类型（avatar修改头像，username修改昵称，area修改地区）（必选，其他参数根据类型选择）",
	"avatar":"string //用户头像地址",
	"username":"string //昵称",
	"areaCode":"int //区县编码（修改地区，只需要传区县编码，不需要传省和城市编码）"
}

返回结果

{
	"code":"int",
	"message":"string",
	"data":{}
}
