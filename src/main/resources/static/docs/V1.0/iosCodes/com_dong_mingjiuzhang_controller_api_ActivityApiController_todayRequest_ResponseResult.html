#import <Foundation/Foundation>;

@interface ActivityRequest : NSObject

//
@property (nonatomic, strong) unknow log;
//
@property (nonatomic, strong) long id;
//活动id
@property (nonatomic, strong) long activityId;
//年级（0其他，1一年级，2二年级，3三年级，4四年级，5五年级，6六年级，7七年级，8八年级）
@property (nonatomic, assign) NSString grade;
//题目类型：1选择题  2填空题
@property (nonatomic, strong) NSInteger type;
//题目内容
@property (nonatomic, assign) NSString title;
//题目图片（json）
@property (nonatomic, assign) NSString titleImage;
//题目选项（json）
@property (nonatomic, assign) NSString requestOption;
//答案内容（json）
@property (nonatomic, assign) NSString answer;
//答案图片（json）
@property (nonatomic, assign) NSString answerImage;
//题目生效时间
@property (nonatomic, assign) NSString gmtStartEffect;
//题目失效时间
@property (nonatomic, assign) NSString gmtEndEffect;
//答对题目所得积分
@property (nonatomic, strong) NSInteger integral;
//是否删除：0否 1是
@property (nonatomic, strong) NSInteger isDeleted;
//添加时间
@property (nonatomic, assign) NSString gmtCreate;
//更新时间
@property (nonatomic, assign) NSString gmtModified;


@end
